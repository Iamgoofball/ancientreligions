# Supported types:
# character, spouse, rivals, friends, liege_titles, vassal_titles, realm_titles, realm_characters, realm_characters_plus, independent_rulers

# Conversion to Druidism
obj_convert_druidism = {
	type = character
	
	potential = {
		age = 16
		is_ruler = yes
		culture_group = celtic
		NOT = { character_flag = refused_druidism }
		NOT = { religion = druidic }
		is_ai = no
	}
	
	allow = {
		learning = 8
		NOT = { trait = incapable }
		NOT = { trait = zealous }
	}
	chance = {
		factor = 100
		
		modifier = {
			factor = 1.25
			trait = cynical
		}
		modifier = {
			factor = 1.25
			trait = patient
		}
		modifier = {
			factor = 1.5
			trait = depressed
		}
		modifier = {
			factor = 0.01
			trait = content
		}
		modifier = {
			factor = 0.01
			trait = slothful
		}
	}
	success = {
		hidden_tooltip = {
			religion = druidic
		}
	}
	abort = {
		character_flag = refused_druidism
		OR = {
			trait = incapable
			trait = zealous
		}
	}
	abort_effect = {
		clr_character_flag = convert_druidism
	}
	effect = {
		piety = 100
		clr_character_flag = convert_druidism
	}
}
