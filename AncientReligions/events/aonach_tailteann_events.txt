# Ruler has died, the Aonach is over
character_event = {
	id = ancrel.1194
	title = TAILTEANNTITLE
	desc = EVTDESCancrel.1194
	picture = GFX_evt_death

	is_triggered_only = yes
	
	trigger = {
		has_character_flag = tailteann_begins
	}
	
	option = {
		name = EVTOPTAancrel.1194
		clr_character_flag = tailteann_begins
		clr_character_flag = winner
		clr_character_flag = second_place
		clr_character_flag = third_place
		clr_character_flag = attending_tailteann
		clr_character_flag = attending_tailteann_event
		clr_character_flag = tailteann_third_place
		clr_character_flag = tailteann_second_place
		clr_character_flag = tailteann_winner
		any_vassal = {
			limit = {
				has_character_flag = attending_tailteann
				NOT = {
					character = PREV
				}
				NOT = {
					has_character_flag = tailteann_third_place
				}
				NOT = {
					has_character_flag = tailteann_second_place
				}
				NOT = {
					has_character_flag = tailteann_winner
				}
			}
			set_character_flag = tailteann_has_ended
			clr_character_flag = attending_tailteann
			clr_character_flag = attending_tailteann_event
			letter_event = {
				id = ancrel.1195
				days = 2
				random = 2
				tooltip = EVTTOOLTIPancrel.1195
			}
		}
		any_vassal = {
			limit = {
				has_character_flag = attending_tailteann
				OR = {
					has_character_flag = tailteann_third_place
					has_character_flag = tailteann_second_place
					has_character_flag = tailteann_winner
				}
			}
			letter_event = {
				id = ancrel.1195
				days = 2
				random = 2
				tooltip = EVTTOOLTIPancrel.1195
			}
			set_character_flag = tailteann_has_ended
			clr_character_flag = tailteann_third_place
			clr_character_flag = tailteann_second_place
			clr_character_flag = tailteann_winner
			clr_character_flag = attending_tailteann
			clr_character_flag = attending_tailteann_event
		}
	}
}

		}

			limit = {

						death_reason = death_battle
