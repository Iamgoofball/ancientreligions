### The Pentarchy

# Patriarchate of Antioch
character_event = {
	id = LoR.200
	desc = EVTDESCLoR.200
	picture = GFX_evt_bishop_byzantine
	
	trigger = {
		has_dlc = "Legacy of Rome"
		OR = {
			has_landed_title = e_latin_empire
			has_landed_title = e_byzantium
			has_landed_title = e_roman_empire
		}
		OR = {
			religion = orthodox
			is_heresy_of = orthodox
		}
		NOT = { has_global_flag = pat_antioch }
		completely_controls = c_antiocheia
	}
	
	option = {
		name = EVTOPTALoR.200
		set_global_flag = pat_antioch
		piety = 100
	}
}

### The Great Schism Mended

# The Schism Mended
narrative_event = {
	id = LoR.210
	title = EVTNAMELoR.210
	desc = EVTDESCLoR.210
	
	picture = GFX_evt_icon
	
	is_triggered_only = yes
	
	option = {
		name = EVTOPTALoR.200
		set_parent_religion = { religion = catholic parent = orthodox }
		set_global_flag = schism_mended
		give_nickname = nick_the_saint
		hidden_tooltip = {
			any_playable_ruler = {
				limit = { 
					religion = catholic
					controls_religion = no
				}
				narrative_event = { id = LoR.211 }
			}
		}
	}
}
narrative_event = {
	id = ancrel.210
	title = EVTNAMELoR.210
	desc = EVTDESCancrel.210
	
	picture = GFX_evt_pope
	
	is_triggered_only = yes
	
	option = {
		name = EVTOPTALoR.200
		set_parent_religion = { religion = orthodox parent = catholic }
		set_global_flag = schism_mended
		give_nickname = nick_the_saint
		if = {
			limit = {
				NOT = { has_landed_title = k_papal_state }
				NOT = { k_papal_state = { holder_scope = { is_vassal_or_below = ROOT } } }
			}
			k_papal_state = { holder_scope = { set_defacto_liege = ROOT } }
		}
		hidden_tooltip = {
			any_playable_ruler = {
				limit = { 
					religion = orthodox
					controls_religion = no
				}
				narrative_event = { id = ancrel.211 }
			}
		}
	}
}
