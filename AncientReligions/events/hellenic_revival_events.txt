# Make Hellenic Paganism the state religion?
character_event = {
	id = ancrel.102
	desc = EVTDESCancrel.102
	picture = GFX_evt_roman_temple
	border = GFX_event_normal_frame_religion

	is_triggered_only = yes
	hide_from = yes

	option = {
		name = EVTOPTAancrel.102
		ai_chance = {
			factor = 50
			modifier = {
				factor = 1.25
				trait = zealous
			}
			modifier = {
				factor = 0.8
				trait = cynical
			}
			modifier = {
				factor = 1.25
				trait = proud
			}
			modifier = {
				factor = 0.8
				trait = humble
			}
			modifier = {
				factor = 0.8
				OR = {
					trait = mastermind_theologian
					trait = scholarly_theologian
					trait = scholar
					trait = mystic
				}
			}
		}
		piety = 50
		if = {
			limit = { NOT = { trait = zealous } }
			random = {
				chance = 50
				add_trait = zealous
			}
		}
		custom_tooltip = EVTTOOLTIPAancrel.102
		hidden_tooltip = {
			any_realm_character  = {
				limit = {
					is_adult = yes
					religion = hellenic_pagan
				}
				opinion = { modifier = opinion_hellenic_state_religion who = ROOT }
			}
			any_realm_character  = {
				limit = {
					is_adult = yes
					NOT = { religion_group = pagan_group }
				}
				opinion = { modifier = opinion_forced_hellenic_state_religion who = ROOT }
			}
			capital_scope = {
				religion = ROOT
			}
			any_courtier = {
				limit = { employer = { character = ROOT } }
				religion = ROOT
			}
			any_vassal = {
				limit = { is_playable = no }
				religion = ROOT
			}
			any_vassal = {
				limit = {
					is_playable = yes
					NOT = { religion = ROOT }
				}
				character_event = { id = ancrel.103 }
			}
		}
	}

	option = {
		name = EVTOPTBancrel.102
		ai_chance = {
			factor = 50
		}
		if = {
			limit = { trait = zealous }
			remove_trait = zealous
		}
		hidden_tooltip = {
			any_realm_character  = {
				limit = {
					is_adult = yes
					religion = hellenic_pagan
				}
				opinion = { modifier = opinion_no_hellenic_state_religion who = ROOT }
			}
			any_realm_character  = {
				limit = {
					is_adult = yes
					NOT = { religion = hellenic_pagan }
				}
				opinion = { modifier = opinion_no_forced_hellenic_state_religion who = ROOT }
			}
		}
	}
}

# Adopt the new Hellenic state religion?
character_event = {
	id = ancrel.103
	desc = EVTDESCancrel.103
	picture = GFX_evt_roman_temple
	border = GFX_event_normal_frame_religion

	is_triggered_only = yes

	option = {
		name = EVTOPTAancrel.103
		ai_chance = {
			factor = 20
			modifier = {
				factor = 0.8
				trait = zealous
			}
			modifier = {
				factor = 1.25
				trait = cynical
			}
			modifier = {
				factor = 0.8
				trait = proud
			}
			modifier = {
				factor = 1.25
				trait = humble
			}
			modifier = {
				factor = 0.8
				religion_authority = 0.6
			}
			modifier = {
				factor = 1.25
				NOT = { religion_authority = 0.4 }
			}
			modifier = {
				factor = 1.25
				opinion = { who = FROM value = 80 }
			}
			modifier = {
				factor = 1.25
				opinion = { who = FROM value = 40 }
			}
			modifier = {
				factor = 0.8
				NOT = { opinion = { who = FROM value = -40 } }
			}
			modifier = {
				factor = 0.8
				NOT = { opinion = { who = FROM value = -80 } }
			}
			modifier = {
				factor = 2.0
				OR = {
					culture = roman
					culture = italian
					culture = greek
				}
			}
			modifier = {
				factor = 3.0
				is_landed = no
				host = { character = FROM }
			}
			modifier = {
				factor = 2.0
				is_landed = yes
				liege = { character = FROM }
			}
			modifier = {
				factor = 2.0
				is_landed = yes
				OR = {
					# free infidel revokation
					crownlaw_title = { has_law = centralization_2 }
					crownlaw_title = { has_law = centralization_3 }
					crownlaw_title = { has_law = centralization_4 }
				}
			}
		}
		religion = FROM
		if = {
			limit = { trait = zealous }
			remove_trait = zealous
		}
		hidden_tooltip = {
			capital_scope = {
				religion = FROM
			}
			any_courtier = {
				limit = { employer = { character = ROOT } }
				religion = FROM
			}
			any_vassal = {
				limit = { is_playable = no }
				religion = FROM
			}
			any_vassal = {
				limit = {
					is_playable = yes
					NOT = { religion = FROM }
				}
				character_event = { id = ancrel.103 }
			}
		}
		reverse_opinion = { modifier = opinion_adopted_hellenic_state_religion who = FROM }
	}

	option = {
		name = EVTOPTBancrel.103
		ai_chance = { factor = 80 }
		piety = 10
		reverse_opinion = { modifier = opinion_rejected_hellenic_state_religion who = FROM }
	}
}
