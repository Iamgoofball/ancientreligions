# You summon your vassals to the celebrations
character_event = {
	id = ancrel.020
	desc = EVTDESCancrel.020
	picture = GFX_evt_viking_throneroom_oldgods
	border = GFX_event_normal_frame_religion
	
	is_triggered_only = yes
	
	hide_from = yes
	
	option = {
		name = EVTOPTAancrel.020
		any_vassal = {
			limit = {
				OR = {
					religion = druidic
					religion = druidic_reformed
					religion = gaelic_pagan
					religion = gaelic_pagan_reformed
				}
				age = 16
				prisoner = no
				war = no
				NOT = { trait = incapable }
			}
			character_event = { id = ancrel.021 tooltip = EVTTOOLTIPancrel.021 }
			hidden_tooltip = { character_event = { id = ancrel.025 days = 10 } }
		}
		any_vassal = {
			limit = {
				OR = {
					religion_group = christian
					religion_group = muslim
				}
				age = 16
				prisoner = no
				war = no
				NOT = { trait = incapable }
			}
			character_event = { id = ancrel.022 tooltip = EVTTOOLTIPancrel.022 }
		}
		hidden_tooltip = { character_event = { id = ancrel.024 days = 10 } }
	}
}

# Non-Druidic Pagans pay to skip celebrations
character_event = {
	id = ancrel.022
	desc = EVTDESCancrel.022
	picture = GFX_evt_viking_throneroom_oldgods
	border = GFX_event_normal_frame_religion
	
	is_triggered_only = yes
	
	hide_from = yes
	
	option = { # Pay
		name = EVTOPTAancrel.022
		trigger = {
			wealth = 20
		}
		FROM = {
			character_event = { id = ancrel.023 tooltip = EVTTOOLTIPancrel.023 }
		}
		wealth = -20
	}
	option = { # Don't pay
		name = EVTOPTBancrel.022
		FROM = {
			character_event = { id = ancrel.056 tooltip = EVTTOOLTIPancrel.056 }
		}
		tooltip = {
			FROM = {
				opinion = {
					modifier = opinion_refused_celtic_feast
					who = ROOT
				}
				any_vassal = {
					limit = { has_character_flag = attending_celtic_feast }
					opinion = {
						modifier = opinion_refused_celtic_feast
						who = ROOT
					}
				}
			}
		}
	}
}

	is_triggered_only = yes
	
	hide_from = yes
	
	option = {
		name = EVTOPTAancrel.024
		hidden_tooltip = {
			if = {
				limit = { 
					any_courtier = {
						prisoner = yes
						host = { character = ROOT }
						age = 16
					}
				}
				random_courtier = {
					limit = { 
						prisoner = yes 
						host = { character = ROOT }
						age = 16
					}
					character_event = { id = ancrel.026 days = 5 }
				}
			}
			if = {
				limit = { 
					NOT = {
						any_courtier = {
							prisoner = yes
							host = { character = ROOT }
							age = 16
						}
					}
				}
				character_event = { id = ancrel.054 days = 5 }
			}
		}
		custom_tooltip = celebrations_begin
	}
}
