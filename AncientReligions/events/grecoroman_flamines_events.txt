namespace = ancrel

# The Flamen Dialis must resign
character_event = {
	id = ancrel.110
	desc = EVTDESCancrel.110
	picture = GFX_evt_council

	trigger = {
		has_minor_title = title_flamen_dialis
		NOT = {
			any_spouse = {
				is_alive = yes
				has_minor_title = title_flaminica_dialis
			}
		}
	}

	option = {
		name = OK
		remove_title = title_flamen_dialis
	}
}
